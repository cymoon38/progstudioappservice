rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    // 공통: 차단 여부 확인 함수
    function isNotBanned() {
      return request.auth != null
        && exists(/databases/$(database)/documents/users/$(request.auth.uid))
        && (
          !('banUntil' in get(/databases/$(database)/documents/users/$(request.auth.uid)).data)
          || get(/databases/$(database)/documents/users/$(request.auth.uid)).data.banUntil <= request.time
        );
    }

    // 공통: 관리자 여부 확인 함수 (role == 'admin' 또는 isAdmin == true)
    function isAdmin(uid) {
      return exists(/databases/$(database)/documents/users/$(uid))
        && (
          get(/databases/$(database)/documents/users/$(uid)).data.role == 'admin'
          || get(/databases/$(database)/documents/users/$(uid)).data.isAdmin == true
        );
    }

    // 게시물 (posts)
    match /posts/{postId} {
      // 읽기는 모든 사용자 허용 (로그인하지 않은 사용자도 작품 볼 수 있음)
      allow read: if true;
      
      // 생성은 인증 및 차단되지 않은 사용자만 허용
      allow create: if isNotBanned();
      
      // 업데이트: 좋아요/댓글 추가는 인증 및 차단되지 않은 사용자만 허용
      allow update: if isNotBanned();
      
      // 삭제: 본인 게시물이거나 관리자(admin)인 경우 허용
      allow delete: if request.auth != null && (
        (
          exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
          get(/databases/$(database)/documents/users/$(request.auth.uid)).data.name == resource.data.author
        ) ||
        isAdmin(request.auth.uid)
      );
    }
    
    // 사용자 정보 (users)
    match /users/{userId} {
      // 읽기는 모든 사용자 허용
      allow read: if true;
      
      // 생성은 인증된 사용자만 허용 (회원가입 시)
      allow create: if request.auth != null && request.auth.uid == userId;
      
      // 업데이트:
      // - 본인인 경우 전체 필드 업데이트 가능
      // - 관리자(admin)는 다른 사용자 문서의 banUntil 필드만 업데이트 가능 (차단/해제)
      // - coins 필드만 업데이트하는 경우 허용 (코인 지급용)
      allow update: if request.auth != null && (
        request.auth.uid == userId ||
        (
          isAdmin(request.auth.uid) &&
          request.resource.data.diff(resource.data).changedKeys().hasOnly(['banUntil'])
        ) ||
        // 코인 지급: coins 필드만 업데이트하는 경우 허용
        (request.resource.data.diff(resource.data).affectedKeys().hasOnly(['coins']) &&
         request.resource.data.coins is int &&
         request.resource.data.coins >= 0)
      );
      allow delete: if request.auth != null && request.auth.uid == userId;
    }
    
    // 알림 (notifications) - 쿼리 권한 포함
    match /notifications/{notificationId} {
      // 개별 문서 읽기: 본인의 알림만
      allow get: if request.auth != null && 
                    request.auth.uid == resource.data.userId;
      
      // 목록 조회 (쿼리): 로그인한 사용자는 자신의 알림만 조회 가능
      // (쿼리에서 userId로 필터링하므로 안전)
      allow list: if request.auth != null;
      
      // 생성: 로그인한 사용자는 알림을 생성할 수 있음
      allow create: if request.auth != null;
      
      // 업데이트: 본인의 알림만 업데이트할 수 있음
      allow update: if request.auth != null && 
                       request.auth.uid == resource.data.userId;
      
      // 삭제: 본인의 알림만 삭제할 수 있음
      allow delete: if request.auth != null && 
                       request.auth.uid == resource.data.userId;
    }
    
    // 코인 내역 (coinHistory)
    match /coinHistory/{historyId} {
      // 읽기: 본인의 내역만 읽을 수 있음
      allow read: if request.auth != null && 
                     resource.data.userId == request.auth.uid;
      
      // 목록 조회 (쿼리): 로그인한 사용자는 자신의 내역만 조회 가능
      // (쿼리에서 userId로 필터링하므로 안전)
      allow list: if request.auth != null;
      
      // 생성: 인증된 사용자가 코인 내역을 생성할 수 있음 (코인 지급 시스템용)
      // - 자신의 내역 생성 가능
      // - 다른 사용자의 내역도 생성 가능 (인기작품 선정 시 코인 지급용)
      // - userId는 유효한 사용자 문서가 존재해야 함
      allow create: if request.auth != null && 
                       request.resource.data.userId is string &&
                       request.resource.data.amount is int &&
                       request.resource.data.type is string &&
                       exists(/databases/$(database)/documents/users/$(request.resource.data.userId));
      
      // 업데이트/삭제: 내역은 수정/삭제 불가
      allow update, delete: if false;
    }
    
    // 미션 (missions)
    match /missions/{missionId} {
      // 읽기는 모든 사용자 허용
      allow read: if true;
      // 생성/수정/삭제는 서버에서만 (클라이언트에서는 불가)
      allow create, update, delete: if false;
    }
    
    // 사용자 미션 (userMissions)
    match /userMissions/{userMissionId} {
      // 개별 문서 읽기: 본인의 미션만 읽을 수 있음 (문서가 존재하지 않을 때도 허용)
      // 문서 ID 형식: userId_missionId
      // 문서가 없거나, 문서 ID가 현재 사용자로 시작하거나, 문서의 userId가 현재 사용자와 일치하면 허용
      allow get: if request.auth != null && 
                   (!resource.exists || 
                    resource.data.userId == request.auth.uid ||
                    userMissionId.matches('^' + request.auth.uid + '_.*'));
      
      // 목록 조회 (쿼리): 로그인한 사용자는 자신의 미션만 조회 가능
      // where('userId', isEqualTo: request.auth.uid) 쿼리를 사용하므로 안전
      // 쿼리 결과의 각 문서는 본인의 것만 반환되어야 함
      allow list: if request.auth != null && 
                    (!resource.exists || resource.data.userId == request.auth.uid);
      
      // 생성: 인증된 사용자가 자신의 미션을 생성할 수 있음
      allow create: if request.auth != null && 
                      request.resource.data.userId == request.auth.uid;
      
      // 업데이트: 인증된 사용자가 자신의 미션을 업데이트할 수 있음
      // 좋아요 3개 누르기 미션은 모든 로그인한 사용자가 참여 가능
      // 문서 ID 형식: userId_missionId이므로 문서 ID로 사용자 확인
      // 또는 기존 문서의 userId가 현재 사용자와 일치하면 허용
      allow update: if request.auth != null && 
                      (userMissionId.matches('^' + request.auth.uid + '_.*') ||
                       (resource.exists && resource.data.get('userId', '') == request.auth.uid));
      
      // 삭제: 본인의 미션만 삭제할 수 있음
      allow delete: if request.auth != null && 
                      (!resource.exists || resource.data.userId == request.auth.uid);
    }
         
         // 추첨 결과 (lotteryResults)
         match /lotteryResults/{date} {
           // 읽기: 모든 인증된 사용자 허용
           allow read: if request.auth != null;
           
           // 생성: 인증된 사용자가 추첨 결과를 생성할 수 있음 (추첨 시스템용)
           allow create: if request.auth != null &&
                            request.resource.data.date is string &&
                            request.resource.data.createdAt is timestamp;
           
           // 수정/삭제: 불가 (추첨 결과는 수정/삭제 불가)
           allow update, delete: if false;
         }
         
         // 캐시 무효화 등 동기화 상태 (syncStatus) - Cloud Functions에서만 쓰기
         match /syncStatus/{docId} {
           allow read: if true;
           allow create, update, delete: if false;
         }
         
         // 기프티콘 상품 (giftcards) - 캐시용
         match /giftcards/{giftcardId} {
           // 읽기: 모든 사용자 허용 (기프티콘 목록 조회용)
           allow read: if true;
           
           // 생성/수정/삭제: 서버에서만 (Cloud Functions에서만)
           allow create, update, delete: if false;
         }
         
         // 구매 내역 (purchases)
         match /purchases/{purchaseId} {
           // 읽기: 본인의 구매 내역만 읽을 수 있음
           allow read: if request.auth != null && 
                         resource.data.userId == request.auth.uid;
           
           // 목록 조회 (쿼리): 로그인한 사용자는 자신의 구매 내역만 조회 가능
           // (쿼리에서 userId로 필터링하므로 안전)
           allow list: if request.auth != null;
           
           // 생성: 서버에서만 (Cloud Functions에서만)
           allow create: if false;
           
           // 업데이트/삭제: 불가
           allow update, delete: if false;
         }
         
         // 보유 기프티콘 (ownedGiftCards)
         match /ownedGiftCards/{cardId} {
           // 읽기: 본인의 보유 기프티콘만 읽을 수 있음
           allow read: if request.auth != null && 
                         resource.data.userId == request.auth.uid;
           
           // 목록 조회 (쿼리): 로그인한 사용자는 자신의 보유 기프티콘만 조회 가능
           // (쿼리에서 userId로 필터링하므로 안전)
           allow list: if request.auth != null;
           
           // 생성: 서버에서만 (Cloud Functions에서만)
           allow create: if false;
           
      // 업데이트: 본인의 보유 기프티콘만, 그리고 차단되지 않은 경우에만 업데이트 가능 (상태 변경 등)
      allow update: if isNotBanned() &&
                      resource.data.userId == request.auth.uid;
           
           // 삭제: 불가
           allow delete: if false;
      }
      
      // 신고 내역 (reports)
      match /reports/{reportId} {
        // 생성: 로그인한 사용자는 신고를 생성할 수 있음
        allow create: if request.auth != null;
        
        // 읽기/목록 조회: 로그인한 사용자만 (UI에서는 관리자만 접근)
        allow read, list: if request.auth != null;
        
        // 수정: 관리자만 dismissed 필드만 변경 가능 (신고 숨기기)
        allow update: if request.auth != null &&
          isAdmin(request.auth.uid) &&
          request.resource.data.diff(resource.data).changedKeys().hasOnly(['dismissed']);

        // 삭제: 불가
        allow delete: if false;
      }
    }
}
