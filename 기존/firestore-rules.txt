Firestore 보안 규칙 설정 가이드
==========================================

⚠️ 중요: 게시물 삭제를 위해 반드시 Firestore 보안 규칙을 설정해야 합니다!

Firebase Console에서 다음 보안 규칙을 설정하세요:

1. Firebase Console 접속: https://console.firebase.google.com
2. 프로젝트 선택: community-b19fb
3. 좌측 메뉴에서 "Firestore Database" 선택
4. "규칙" 탭 클릭
5. 아래 규칙을 복사하여 붙여넣기:

--- 권장 규칙 (읽기는 모두 허용, 좋아요/댓글은 모든 인증된 사용자, 삭제는 작성자만) ---
rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    // 게시물 (posts)
    match /posts/{postId} {
      // 읽기는 모든 사용자 허용 (로그인하지 않은 사용자도 작품 볼 수 있음)
      allow read: if true;
      
      // 생성은 인증된 사용자만 허용
      allow create: if request.auth != null;
      
      // 업데이트: 좋아요/댓글 추가는 모든 인증된 사용자 허용, 삭제는 작성자만
      allow update: if request.auth != null;
      
      // 삭제는 본인이 작성한 게시물만 허용
      allow delete: if request.auth != null && (
        // 사용자 정보에서 이름 가져와서 비교
        exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.name == resource.data.author
      );
    }
    
    // 사용자 정보 (users)
    match /users/{userId} {
      // 읽기는 모든 사용자 허용
      allow read: if true;
      
      // 생성은 인증된 사용자만 허용 (회원가입 시)
      allow create: if request.auth != null && request.auth.uid == userId;
      
      // 수정/삭제는 본인만 허용
      allow update, delete: if request.auth != null && request.auth.uid == userId;
    }
  }
}

--- 권장 규칙 (좋아요/댓글은 모든 인증된 사용자, 삭제는 작성자만) ---
rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    // 게시물 (posts)
    match /posts/{postId} {
      // 읽기는 모든 사용자 허용 (로그인하지 않은 사용자도 작품 볼 수 있음)
      allow read: if true;
      
      // 생성은 인증된 사용자만 허용
      allow create: if request.auth != null;
      
      // 업데이트: 좋아요/댓글 추가는 모든 인증된 사용자 허용
      allow update: if request.auth != null;
      
      // 삭제는 본인이 작성한 게시물만 허용
      allow delete: if request.auth != null && (
        exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.name == resource.data.author
      );
    }
    
    // 사용자 정보 (users)
    match /users/{userId} {
      // 읽기는 모든 사용자 허용
      allow read: if true;
      
      // 생성은 인증된 사용자만 허용 (회원가입 시)
      allow create: if request.auth != null && request.auth.uid == userId;
      
      // 수정/삭제는 본인만 허용
      allow update, delete: if request.auth != null && request.auth.uid == userId;
    }
  }
}

--- 더 간단한 규칙 (개발용 - 빠른 테스트) ---
rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    // 모든 인증된 사용자가 읽기/쓰기/수정 가능, 삭제는 제한 (개발용)
    match /{document=**} {
      allow read: if true;  // 모든 사용자 읽기 허용
      allow create, update: if request.auth != null;  // 인증된 사용자만 생성/수정 허용
      allow delete: if request.auth != null;  // 인증된 사용자만 삭제 허용 (개발용)
    }
  }
}

6. "게시" 버튼 클릭하여 규칙 저장

⚠️ 주의사항:
- 첫 번째 규칙(권장 규칙)을 사용하면 모든 인증된 사용자가 좋아요/댓글을 추가할 수 있습니다.
- 게시물 삭제는 작성자만 가능합니다.
- 위 간단한 규칙은 빠른 테스트용입니다. 프로덕션 환경에서는 첫 번째 규칙 사용을 권장합니다.
- 규칙을 변경한 후 바로 적용되지만, 최대 1-2분 정도 걸릴 수 있습니다.
- 규칙 저장 후 브라우저를 새로고침하고 다시 테스트해보세요.

추가 참고:
- 게시물 삭제 시 작성자 이름(author)과 현재 로그인한 사용자의 이름이 일치해야 합니다.
- 사용자 이름은 Firestore의 users 컬렉션에 저장된 name 필드와 일치해야 합니다.
- 좋아요와 댓글 추가는 모든 로그인된 사용자가 할 수 있습니다.
